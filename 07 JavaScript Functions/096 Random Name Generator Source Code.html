    <script>
        var adj = ["super", "wonderful", "bad", "angry", "careful"];

        function myFun() {

            var nameAdj = Math.floor(Math.random() * adj.length);
            output.innerHTML += adj[nameAdj] + " " + question + "<br>";
        }
